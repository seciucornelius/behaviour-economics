	
	#include <iostream>
	#include <vector>
	#include <cstdlib>
	#include <ctime>
	#include <iomanip>
	#include <algorithm>
	
	using namespace std;
	
	int main() {
	    const int numAgents = 100;
	    const int numDays = 365;
	    const double baseProb = 0.4;
	    const double nudgeEffect = 0.2;
	    const double socialFactor = 0.001;
	
	    srand(time(0));
	
	    vector<vector<bool>> decisions(numDays, vector<bool>(numAgents));
	    vector<double> dailyRecycleRate(numDays);
	
	    for (int day = 0; day < numDays; ++day) {
	        int recycledToday = 0;
	        double socialInfluence = 0.0;
	
	        if (day > 0) {
	            int prevDayRecyclers = 0;
	            for (int a = 0; a < numAgents; ++a) {
	                if (decisions[day - 1][a]) prevDayRecyclers++;
	            }
	            socialInfluence = socialFactor * prevDayRecyclers;
	        }
	
	        for (int a = 0; a < numAgents; ++a) {
	            double recycleChance = baseProb + nudgeEffect + socialInfluence;
	            double randomDraw = static_cast<double>(rand()) / RAND_MAX;
	
	            if (randomDraw < recycleChance) {
	                decisions[day][a] = true;
	                recycledToday++;
	            } else {
	                decisions[day][a] = false;
	            }
	        }
	
	        dailyRecycleRate[day] = static_cast<double>(recycledToday) / numAgents;
	    }
	
	    // Afișare rată zilnică (opțional)
	    cout << "Zi\tRata reciclare\n";
	    for (int day = 0; day < numDays; ++day) {
	        cout << setw(3) << day + 1 << "\t" << fixed << setprecision(2)
	             << dailyRecycleRate[day] * 100 << "%" << endl;
	    }
	
	    // Calcul medii lunare
	    cout << "\n--- Medii lunare ---\n";
	    for (int month = 0; month < 12; ++month) {
	        int startDay = month * 30;
	        int endDay = min(startDay + 30, numDays);  // Ultima lună poate avea < 30 zile
	        double sum = 0.0;
	        for (int d = startDay; d < endDay; ++d) {
	            sum += dailyRecycleRate[d];
	        }
	        double monthlyAvg = sum / (endDay - startDay);
	        cout << "Luna " << month + 1 << ": " << fixed << setprecision(2)
	             << monthlyAvg * 100 << "%" << endl;
	    }
	
	    return 0;
	}
